水平垂直居中


情景一:已知宽高


        .box-container1 {
            position: relative;
        }

        .box1 {
            width: 200px;
            height: 100px;

            position: absolute;
            left: 50%; /* 配合margin-left的负值实现水平居中 */
            margin-left: -110px; /* 值的大小等于元素宽度的一半 */
            top: 50%; /* 配合margin-top的负值实现垂直居中 */
            margin-top: -50px; /* 值的大小等于元素高度的一半 */
        }
child-box

每次刷新页面,其宽度和高度都会动态地改变:


        let $box2 = document.getElementsByClassName('box2')[0],
            $box3 = document.getElementsByClassName('box3')[0];

        $box2.style.width  = getRandomInt(100,300) + 'px';
        $box2.style.height = getRandomInt(50,200) + 'px';

        $box3.style.width  = getRandomInt(100,300) + 'px';
        $box3.style.height = getRandomInt(50,200) + 'px';

        /**
         * 获取随机数
         * @param   {int}  min  范围的最小值
         * @param   {int}  max  范围的最大值
         * @return  {int}  min  随机整数
         */
        function getRandomInt(min, max) {
            return Math.floor(Math.random() * (max - min + 1)) + min;
        }
